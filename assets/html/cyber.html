# Simulasi validasi password kuat
import re

sandi = "Cyb3rKuat!"
if re.match(r"^(?=.*[A-Z])(?=.*[a-z])(?=.*\\d)(?=.*[!@#\$%]).{8,}$", sandi):
    print("Sandi kuat")
else:
    print("Sandi lemah")
      
# Simulasi deteksi brute-force
percobaan = 0
max_login = 5
while percobaan < max_login:
    sandi = input("Masukkan sandi: ")
    if sandi == "admin123":
        print("Login berhasil!")
        break
    else:
        print("Salah, coba lagi.")
        percobaan += 1
